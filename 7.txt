class Solution {
    public int reverse(int x) {
        if(x == 0) {
            return 0;
        }
        else {
            boolean isNeg = (x > 0);
            x = Math.abs(x);
            String rev;
            rev = isNeg ? "" : "-";
            while(x > 0){
                rev += x %10;
                x/=10;
            }
            return Integer.parseInt(rev);
        }
    }
}
