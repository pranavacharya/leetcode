class Solution {
    public int searchInsert(int[] nums, int target) {
        int high = nums.length - 1;
        int low = 0;
        int mid = 0;
        int index = -1;
        while (low < high) {
            mid = (high + low) / 2;
            if (nums[mid] == target) {
                index = mid;
                break;
            }
            else if (nums[mid] > target)
                high = mid - 1;
            else 
                low = mid + 1;
        }
        return index;
    }
}
