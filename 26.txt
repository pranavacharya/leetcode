class Solution {
    public int removeDuplicates(int[] nums) {
        int i = 0;
        int j = 1;
        int n = nums.length;
        while (j < n) {
            if (nums[i] == nums[j]) {
                nums[j] = nums[n-1];
                n--;
            }
            else {
                i++;
                j++;
            }
        }
        return n;
    }
}
